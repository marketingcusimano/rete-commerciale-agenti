createClientiChart() { const ctx = document.getElementById('clientiChart'); if (!ctx) return; // Destroy existing chart if it exists if (this.clientiChart) { this.clientiChart.destroy(); } const ltClienti = this.csvData.latina.clientiCorrente; const rmClienti = this.csvData.roma.clienti; const ltPrevClienti = this.csvData.latina.clientiPrecedenti; // Calculate dynamic max value const maxValue = Math.max(ltClienti, rmClienti, ltPrevClienti) + 5; this.clientiChart = new Chart(ctx, { type: 'bar', data: { labels: ['Latina (LT)', 'Roma (RM)'], datasets: [ { label: '2024', data: [ltPrevClienti, 0], // RM non ha dati 2024 backgroundColor: '#e2e8f0', borderColor: '#cbd5e1', borderWidth: 1, borderRadius: 8, borderSkipped: false, maxBarThickness: 80 }, { label: '2025', data: [ltClienti, rmClienti], backgroundColor: ['#8b5cf6', '#22d3ee'], borderColor: ['#7c3aed', '#06b6d4'], borderWidth: 2, borderRadius: 8, borderSkipped: false, maxBarThickness: 80 } ] }, options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { display: true, position: 'top', labels: { padding: 20, font: { size: 12, weight: '600' }, color: '#374151', usePointStyle: true, pointStyle: 'rect' } }, tooltip: { backgroundColor: '#1f2937', titleColor: '#f9fafb', bodyColor: '#f9fafb', borderColor: '#374151', borderWidth: 1, cornerRadius: 8, callbacks: { title: (context) => `${context[0].label}`, label: (context) => `${context.dataset.label}: ${context.raw} clienti` } } }, scales: { y: { beginAtZero: true, max: maxValue, ticks: { stepSize: Math.ceil(maxValue / 6), color: '#6b7280', font: { size: 11, weight: '500' }, callback: (value) => `${value}` }, grid: { color: '#f3f4f6', drawBorder: false }, border: { display: false } }, x: { ticks: { color: '#374151', font: { size: 12, weight: '600' } }, grid: { display: false }, border: { display: false } } }, animation: { duration: 2000, easing: 'easeOutQuart' }, interaction: { intersect: false, mode: 'index' } } }); }
